favicolor

Spruce up your site with a color-coordinating favicon!

your-site.io
Tap on a color Click on a color

Example:

var icon = document.querySelector('[rel=icon]');
var favicolor = require('favicolor');
favicolor(icon, 'hotpink');     /* any valid web color */
favicolor(icon, [133, 20, 75]); /* an RGB array */
favicolor(icon, 255, 99, 71);   /* or individual values */
